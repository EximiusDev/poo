class Triang{
	float base, altura;
public:
	Triang(float b, float h):base(b),altura(h){}
	void Datos(float b, float h){base=b;altura=h;}
	float Area(){return base*altura/2.0;}
	friend float AreaPrisma(float b,float htr, float hre);
};
class Rectang{
	float base, altura;
public:
	Rectang(float b, float h):base(b),altura(h){}
	void Datos(float b, float h){base=b;altura=h;}
	float Area(){return base*altura;}
	friend float AreaPrisma(float b,float htr, float hre);
};

class PrismaTriang{     //creo que es la mejor opcion
	Triang t;
	Rectang r;
public:
	PrismaTriang(float bt,float ht, float hr):t(bt,ht),r(bt,hr){}  //la base del triang es igual a la base del rectang
    float Area(){return 2*t.Area() + 3*r.Area();}
};


float AreaPrisma(float b,float htr, float hre){   //con amistad
  Triang t(b,htr);
  Rectang r(b,hre);
  return 2*r.base*t.altura/2 + 3*r.base*r.altura;
}
  
float AreaPrismaSinAmistad(float b,float htr, float hre){
	  Triang t(b,htr);
	  Rectang r(b,hre);
	  return 2*t.Area() + 3*r.Area();
}
